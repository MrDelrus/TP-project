cmake_minimum_required(VERSION 3.15)
project(TP_project)

set(CMAKE_CXX_STANDARD 20)

add_executable(TP_project main.cpp)
add_library(lib_st SHARED libST.cpp)
set_target_properties(lib_st PROPERTIES LINKER_LANGUAGE CXX)
add_library(in_out SHARED in_out.cpp)
set_target_properties(in_out PROPERTIES LINKER_LANGUAGE CXX)
add_library(lib_c SHARED Chat.cpp)
set_target_properties(lib_c PROPERTIES LINKER_LANGUAGE CXX)
add_library(lib_ptg SHARED libPTG.cpp)
set_target_properties(lib_ptg PROPERTIES LINKER_LANGUAGE CXX)
target_link_libraries(TP_project lib_c)
target_link_libraries(TP_project lib_st)
target_link_libraries(TP_project lib_ptg)
target_link_libraries(TP_project in_out)
